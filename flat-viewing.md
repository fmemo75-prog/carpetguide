    <div class="section-label">Origins</div>
    <h2>Woven flat, <em>lasting forever</em></h2>
    <p>
      Kilim is a pileless flatweave textile produced across Anatolia, the Caucasus, and Central Asia
      for at least three thousand years. Unlike pile rugs — where knotted loops create surface texture
      — a kilim is constructed entirely through interlacing weft and warp threads, leaving no pile at all.
    </p>
    <p>
      The result is a thinner, lighter, reversible textile whose pattern appears identically on both
      faces. The geometric vocabulary of kilims — diamonds, hexagons, stepped triangles, ram's horns —
      emerges directly from the structural logic of the slit tapestry technique.
    </p>

function goTo(tech, n) {
  const d = DATA[tech];
  d.cur = n;

  // hide all steps
  const svg = document.getElementById('svg-' + tech);
  const prefix = tech === 'weft' ? 'ws' : 'ss';
  [0,1,2,3].forEach(i => {
    const el = svg.querySelector('#' + prefix + i);
    if (el) { el.style.opacity = i === n ? '1' : '0'; el.style.transition = 'opacity .35s'; }
  });

  // animate paths in current step
  const cur = svg.querySelector('#' + prefix + n);
  if (cur) {
    cur.querySelectorAll('.anim-path').forEach(p => {
      const m = p.style.cssText.match(/--len:\s*([\d.]+)/);
      const len = m ? parseFloat(m[1]) : 300;
      if (len === 0) return;
      p.style.animation = 'none';
      void p.offsetWidth;
      p.style.strokeDasharray = len;
      p.style.strokeDashoffset = len;
      setTimeout(() => {
        p.style.animation = `drawPath .9s cubic-bezier(.4,0,.2,1) forwards`;
        p.style.strokeDashoffset = 0;
      }, 30);
    });
  }

  // slit step 1 extras
  if (tech === 'slit' && n === 1) {
    setTimeout(() => {
      const gap = document.getElementById('slit-gap');
      const lbl = document.getElementById('slit-label');
      if (gap) gap.style.opacity = '1';
      if (lbl) lbl.style.opacity = '1';
      // show return rows
      cur.querySelectorAll('.anim-path[opacity="0"]').forEach((p, i) => {
        p.style.opacity = '1';
        p.style.animation = 'none';
        void p.offsetWidth;
        const m = p.style.cssText.match(/--len:\s*([\d.]+)/);
        const len = m ? parseFloat(m[1]) : 200;
        p.style.strokeDasharray = len;
        p.style.strokeDashoffset = len;
        p.style.animation = `drawPath .8s ${.5 + i*.15}s cubic-bezier(.4,0,.2,1) forwards`;
      });
    }, 600);
  }

  // tabs
  const tabs = document.querySelectorAll('#tabs-' + tech + ' .step-tab');
  tabs.forEach((b, i) => b.classList.toggle('active', i === n));

  // dots
  const dots = document.querySelectorAll('#dots-' + tech + ' .pdot');
  dots.forEach((d, i) => d.classList.toggle('active', i === n));

  // buttons
  document.getElementById('prev-' + tech).disabled = n === 0;
  document.getElementById('next-' + tech).disabled = n === 3;

  // desc
  const descEl = document.getElementById('desc-' + tech);
  const box = descEl.parentElement;
  box.style.animation = 'none';
  void box.offsetWidth;
  box.style.animation = 'fadeUp .4s ease forwards';
  descEl.innerHTML = DATA[tech].descs[n];
}

function next(tech) { const d = DATA[tech]; if (d.cur < 3) goTo(tech, d.cur + 1); }
function prev(tech) { const d = DATA[tech]; if (d.cur > 0) goTo(tech, d.cur - 1); }

function switchTech(tech) {
  document.querySelectorAll('.tech-panel').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.tech-btn').forEach((b, i) =>
    b.classList.toggle('active', (i === 0 && tech === 'weft') || (i === 1 && tech === 'slit')));
  document.getElementById('panel-' + tech).classList.add('active');
}

  document.querySelectorAll('.anatomy-view').forEach(v => v.classList.remove('active'));
  document.querySelectorAll('.atab').forEach((b, i) =>
    b.classList.toggle('active', (i === 0 && id === 'weave') || (i === 1 && id === 'slit')));

}
